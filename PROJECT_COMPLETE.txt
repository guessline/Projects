📈 ФОРМУЛЫ МЕТРИК (Quick Reference)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   MAPE:  =IFERROR(AVERAGE(IF(ISNUMBER(rngPctAbsE);rngPctAbsE));NA())
   WAPE:  =IF(SUM(rngY)>0;SUM(rngAbsE)/SUM(rngY);NA())
   MAE:   =IFERROR(AVERAGE(IF(rngAbsE<>"";rngAbsE));NA())
   RMSE:  =IFERROR(SQRT(AVERAGE((rngE)^2));NA())
   Bias:  =IFERROR(AVERAGE(IF(rngE<>"";rngE));NA())
